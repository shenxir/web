```javascript
// frontend/src/stores/auth.js
export const useAuthStore = defineStore('auth', () => {
  const currentUser = ref(getItem('currentUser'))

  const isLoggedIn = computed(() => currentUser.value !== null)
  const isAdmin = computed(() => currentUser.value?.role === 'admin')

  async function login(username, password) {
    try {
      const data = await api.post('/auth/login', { username, password })
      if (data.success) {
        currentUser.value = data.user
        setItem('currentUser', data.user)
        return { success: true, message: data.message }
      } else {
        return { success: false, message: data.message }
      }
    } catch (e) {
      return { success: false, message: '登录失败，请重试' }
    }
  }

  function logout() {
    currentUser.value = null
    removeItem('currentUser')
  }

  return { currentUser, isLoggedIn, isAdmin, login, logout }
})
```

```java
// UserService.java
@Service
public class UserService {

    @Autowired
    private UserMapper userMapper;

    public Map<String, Object> login(String username, String password) {
        Map<String, Object> result = new HashMap<>();

        // 查询用户
        User user = userMapper.getUserByUsername(username);
        if (user == null) {
            result.put("success", false);
            result.put("message", "用户名不存在");
            return result;
        }

        // 验证密码
        if (!user.getPassword().equals(password)) {
            result.put("success", false);
            result.put("message", "密码错误");
            return result;
        }

        // 检查用户状态
        if (user.getStatus() != 1) {
            result.put("success", false);
            result.put("message", "账户已被禁用");
            return result;
        }

        result.put("success", true);
        result.put("message", "登录成功");
        result.put("user", user);
        return result;
    }
}
```

```java
        result.put("message", "注册成功");
```

#### （3）路由守卫实现

```javascript
// frontend/src/router/guards.js
export function setupRouterGuards(router) {
  router.beforeEach((to, from, next) => {
    const authStore = useAuthStore()

    // 设置页面标题
    if (to.meta.title) {
      document.title = `${to.meta.title} - 校园失物招领系统`
    }

    // 需要登录的页面
    if (to.meta.requiresAuth && !authStore.isLoggedIn) {
      next({
        name: 'Login',
        query: { redirect: to.fullPath }
      })
      return
    }

    // 需要管理员权限的页面
    if (to.meta.requiresAdmin && !authStore.isAdmin) {
      ElMessage.error('没有权限访问该页面')
      next({ name: 'Home' })
      return
    }

    // 游客只能访问的页面（登录/注册）
    if (to.meta.guestOnly && authStore.isLoggedIn) {
      next({ name: 'Home' })
      return
    }

    next()
  })
}
```

#### （3）后端物品发布接口

```java
// ItemController.java
@Transactional
public Map<String, Object> publishItem(Item item, List<String> imageUrls) {
    Map<String, Object> result = new HashMap<>();

    // 检查重复发布（24小时内相同标题）
    int duplicate = itemMapper.checkDuplicate(item.getUserId(), item.getTitle());
    if (duplicate > 0) {
        result.put("success", false);
        result.put("message", "您在24小时内已发布过相同标题的物品");
        return result;
    }

    // 设置默认值
    LocalDate today = LocalDate.now();
    item.setPublishDate(today.format(DATE_FMT));
    item.setExpiryDate(today.plusDays(7).format(DATE_FMT));
    item.setStatus("pending");

    // 插入物品记录
    itemMapper.insertItem(item);

    // 插入图片记录
    if (imageUrls != null && !imageUrls.isEmpty()) {
        for (int i = 0; i < imageUrls.size(); i++) {
            ItemImage image = new ItemImage();
            image.setItemId(item.getId());
            image.setImageUrl(imageUrls.get(i));
            image.setSortOrder(i + 1);
            imageMapper.insertImage(image);
        }
    }

    result.put("success", true);
    result.put("message", "发布成功");
    result.put("item", item);
    return result;
}
```

```java
    SELECT i.*, u.real_name AS user_name,
```

```java
// ClaimService.java
@Transactional
public Map<String, Object> createClaim(Claim claim) {
    Map<String, Object> result = new HashMap<>();

    // 检查是否已认领过
    boolean alreadyClaimed = claimMapper.checkDuplicateClaim(
        claim.getItemId(), claim.getClaimerId());
    if (alreadyClaimed) {
        result.put("success", false);
        result.put("message", "您已申请认领过该物品");
        return result;
    }

    // 检查是否是物品发布者
    Item item = itemMapper.getItemById(claim.getItemId());
    if (item.getUserId().equals(claim.getClaimerId())) {
        result.put("success", false);
        result.put("message", "不能认领自己发布的物品");
        return result;
    }

    // 创建认领记录
    claim.setStatus("pending");
    int rows = claimMapper.insertClaim(claim);

    if (rows > 0) {
        result.put("success", true);
        result.put("message", "认领申请已提交，请等待管理员审核");
    } else {
        result.put("success", false);
        result.put("message", "申请失败，请重试");
    }
    return result;
}
```

#### （3）管理员认领审核

```java
// ClaimService.java
@Transactional
public Map<String, Object> reviewClaim(Integer claimId, String status,
                                        String adminNote, Integer reviewedBy) {
    Map<String, Object> result = new HashMap<>();

    // 更新认领状态
    int rows = claimMapper.reviewClaim(claimId, status, adminNote, reviewedBy);

    if (rows > 0) {
        // 如果审核通过，更新物品状态为已认领
        if ("approved".equals(status)) {
            Claim claim = claimMapper.getClaimById(claimId);
            itemMapper.updateItemStatus(claim.getItemId(), "claimed");
        }

        result.put("success", true);
        result.put("message", "审核完成");
    } else {
        result.put("success", false);
        result.put("message", "审核失败");
    }
    return result;
}
```

```javascript
// frontend/src/api/index.js
const api = axios.create({

api.interceptors.request.use((config) => {

api.interceptors.response.use(
  (response) => response.data,

        case 403:
          ElMessage.error('没有权限执行此操作')
          break
        case 500:
          ElMessage.error('服务器错误')
          break
      }
    }
    return Promise.reject(error)
  }
)
```

### 3.4.2 文件上传处理

后端文件上传接口，支持图片存储：

```java
// ItemController.java
@PostMapping("/upload")
public Map<String, Object> uploadImage(@RequestParam("file") MultipartFile file) {
    Map<String, Object> result = new HashMap<>();

    try {
        String originalFilename = file.getOriginalFilename();
        String extension = originalFilename.substring(originalFilename.lastIndexOf("."));

        // 生成唯一文件名
        String timestamp = new SimpleDateFormat("yyyyMMddHHmmss").format(new Date());
        String filename = "item_" + timestamp + "_" +
                         UUID.randomUUID().toString().substring(0, 8) + extension;

        // 保存文件
        String projectDir = System.getProperty("user.dir");
        String uploadPath = projectDir + File.separator + "picture";
        File dir = new File(uploadPath);
        if (!dir.exists()) {
            dir.mkdirs();
        }

        File dest = new File(dir, filename);
        file.transferTo(dest);

        result.put("success", true);
        result.put("url", "/picture/" + filename);
        result.put("originalName", originalFilename);
    } catch (IOException e) {
        result.put("success", false);
        result.put("message", "上传失败: " + e.getMessage());
    }

    return result;
}
```

## 3.6 本章小结

本章详细介绍了校园失物招领系统的开发与实现过程。系统采用前后端分离架构，后端基于Spring Boot框架，前端使用Vue 3 + Element Plus技术栈。

**主要实现的功能模块包括：**

1. **用户认证模块**：实现用户注册、登录、权限控制
2. **物品发布模块**：支持多图上传、表单验证、重复发布检测
3. **物品搜索模块**：支持关键词、分类、地点多维度筛选
4. **认领管理模块**：实现认领申请、管理员审核流程
5. **管理后台模块**：数据统计、物品管理、公告管理
6. **自动过期处理**：定时检查并更新过期物品状态

**关键技术点：**

- 使用Pinia进行前端状态管理
- 使用MyBatis实现灵活的SQL映射
- 使用@Transactional确保事务一致性
- 使用Axios拦截器统一处理请求/响应
- 使用Vue Router守卫实现路由权限控制

通过以上功能的实现，系统能够满足校园师生的失物招领需求，提供便捷的信息发布和认领服务。
